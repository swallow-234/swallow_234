      const tabs = document.querySelectorAll('.tab');

------AI解釋::------
function showTab(index) {
  // 取得所有 class 為 'tab' 的元素（通常是分頁標籤）
  const tabs = document.querySelectorAll('.tab');

  // 取得所有 class 為 'tab-content' 的元素（通常是分頁內容）
  const contents = document.querySelectorAll('.tab-content');

  // 針對每一個 tab 以及對應的內容做迴圈
  tabs.forEach((tab, i) => {
    // 如果目前迴圈的索引值 i 等於要顯示的 index，則加上 'active' class，否則移除
    tab.classList.toggle('active', i === index);

    // 對應的內容也做一樣的處理
    contents[i].classList.toggle('active', i === index);
  });
}
功能說明
用途：這個函式通常用於多個分頁（Tab）切換內容的網頁元件。
參數：index 代表你要顯示哪一個分頁（從 0 開始）。
流程：
        取得所有的 .tab 和 .tab-content 元素。
        用 forEach 逐一處理每一個 tab 和內容。
        只有當前選中的 tab 和內容（i === index）會加上 active class，其他都移除 active。
效果：
被選中的分頁和內容會有 active 樣式（通常用 CSS 控制顯示／隱藏）。
其他分頁和內容則不會有 active 樣式（通常被隱藏）。
